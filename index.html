function sendCommand() { const userInput = document.getElementById('user-input').value; fetch('/process-command', { method: 'POST', headers: { 'Content-Type': 'application/json' }, body: JSON.stringify({ command: userInput }) }) .then(response => response.json()) .then(data => { const gameOutput = document.getElementById('game-output'); gameOutput.innerHTML += `

${data.response}

`; document.getElementById('user-input').value = ''; }) .catch(error => console.error('Error:', error)); }
